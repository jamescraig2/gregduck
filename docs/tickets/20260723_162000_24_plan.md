## Phase 2: Core Component & Layout Implementation

1. **Glass UI Primitives**:
   - Create `components/ui/GlassCard.tsx`:
     ```tsx
     import React from 'react';
     export interface GlassCardProps extends React.HTMLAttributes<HTMLDivElement> {
       children: React.ReactNode;
       className?: string;
       hoverEffect?: boolean;
     }
     export const GlassCard: React.FC<GlassCardProps> = ({
       children,
       className = '',
       hoverEffect = true,
       ...props
     }) => (
       <div
         className={`glass-card ${hoverEffect ? '' : 'no-hover'} ${className}`.trim()}
         {...props}
       >
         {children}
       </div>
     );
     ```
   - Create `components/ui/GlassPanel.tsx`:
     ```tsx
     import React from 'react';
     export interface GlassPanelProps extends React.HTMLAttributes<HTMLDivElement> {
       children: React.ReactNode;
       className?: string;
     }
     export const GlassPanel: React.FC<GlassPanelProps> = ({
       children,
       className = '',
       ...props
     }) => (
       <div className={`glass-panel ${className}`.trim()} {...props}>
         {children}
       </div>
     );
     ```
   - Create `components/ui/Skeleton.tsx`:
     ```tsx
     import React from 'react';
     export interface SkeletonProps extends React.HTMLAttributes<HTMLDivElement> {
       width?: string | number;
       height?: string | number;
       borderRadius?: string | number;
       className?: string;
     }
     export const Skeleton: React.FC<SkeletonProps> = ({
       width,
       height,
       borderRadius,
       className = '',
       style,
       ...props
     }) => {
       const customStyles: React.CSSProperties = {
         width: typeof width === 'number' ? `${width}px` : width,
         height: typeof height === 'number' ? `${height}px` : height,
         borderRadius: typeof borderRadius === 'number' ? `${borderRadius}px` : borderRadius,
         ...style,
       };
       return (
         <div
           className={`skeleton-loader ${className}`.trim()}
           style={customStyles}
           aria-label="Loading..."
           role="status"
           {...props}
         />
       );
     };
     ```
   - Update [`components/index.ts`](file:///Users/jamescraig/gh/gregduck/components/index.ts#L1-L2) to export all primitives.
